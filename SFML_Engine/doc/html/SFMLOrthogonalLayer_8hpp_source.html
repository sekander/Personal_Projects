64  MapLayer(const tmx::Map& map, std::size_t idx)
65  {
66  const auto& layers = map.getLayers();
67  if (map.getOrientation() == tmx::Orientation::Orthogonal &&
68  idx < layers.size() && layers[idx]->getType() == tmx::Layer::Type::Tile)
69  {
70  //round the chunk size to the nearest tile
71  const auto tileSize = map.getTileSize();
72  m_chunkSize.x = std::floor(m_chunkSize.x / tileSize.x) * tileSize.x;
73  m_chunkSize.y = std::floor(m_chunkSize.y / tileSize.y) * tileSize.y;
74  m_MapTileSize.x = map.getTileSize().x;
75  m_MapTileSize.y = map.getTileSize().y;
76  const auto& layer = layers[idx]->getLayerAs<tmx::TileLayer>();
77  createChunks(map, layer);
78 
79  auto mapSize = map.getBounds();
80  m_globalBounds.width = mapSize.width;
81  m_globalBounds.height = mapSize.height;
82  }
83  else
84  {
85  std::cout << "Not a valid orthogonal layer, nothing will be drawn." << std::endl;
86  }
87  }

123  {
124  for (auto& c : m_visibleChunks)
125  {
126  for (AnimationState& as : c->getActiveAnimations())
127  {
128  as.currentTime += elapsed;

132  tile.flipFlags = 0; // TODO: get flipFlags from original tmx::TileLayer::Tile
133 
134  std::uint32_t animTime = 0;

140  tile.ID = frame.tileID;

143  as.currentTime = sf::milliseconds(0);

145  }

186  Chunk(const tmx::TileLayer& layer, std::vector<const tmx::Tileset*> tilesets,
187  const sf::Vector2f& position, const sf::Vector2f& tileCount, const sf::Vector2u& tileSize,
188  std::size_t rowSize, TextureResource& tr, const std::map<std::uint32_t, tmx::Tileset::Tile>& animTiles)
189  : m_animTiles(animTiles)
190  {
191  setPosition(position);
192  layerOpacity = static_cast<sf::Uint8>(layer.getOpacity() / 1.f * 255.f);
193  sf::Color vertColour = sf::Color(200 ,200, 200, layerOpacity);
194  auto offset = layer.getOffset();
195  layerOffset.x = offset.x;
196  layerOffset.x = offset.y;
197  chunkTileCount.x = tileCount.x;
198  chunkTileCount.y = tileCount.y;
199  mapTileSize = tileSize;
200  const auto& tileIDs = layer.getTiles();
201 
202  //go through the tiles and create all arrays (for latter manipulation)
203  for (const auto& ts : tilesets)
204  {
205  if(ts->getImagePath().empty())
206  {
207  tmx::Logger::log("This example does not support Collection of Images tilesets", tmx::Logger::Type::Info);
208  tmx::Logger::log("Chunks using " + ts->getName() + " will not be created", tmx::Logger::Type::Info);
209  continue;
210  }
211  m_chunkArrays.emplace_back(std::make_unique<ChunkArray>(*tr.find(ts->getImagePath())->second, *ts));
212  }
213  std::size_t xPos = static_cast<std::size_t>(position.x / tileSize.x);
214  std::size_t yPos = static_cast<std::size_t>(position.y / tileSize.y);
215  for (auto y = yPos; y < yPos + tileCount.y; ++y)
216  {
217  for (auto x = xPos; x < xPos + tileCount.x; ++x)
218  {
219  auto idx = (y * rowSize + x);
220  m_chunkTileIDs.emplace_back(tileIDs[idx]);
221  m_chunkColors.emplace_back(vertColour);
222  }
223  }
224  generateTiles(true);
225  }

227  void generateTiles(bool registerAnimation = false)
228  {
229  if (registerAnimation)
230  {
231  m_activeAnimations.clear();
232  }
233  for (const auto& ca : m_chunkArrays)
234  {
235  sf::Uint32 idx = 0;
236  std::size_t xPos = static_cast<std::size_t>(getPosition().x / mapTileSize.x);
237  std::size_t yPos = static_cast<std::size_t>(getPosition().y / mapTileSize.y);
238  for (auto y = yPos; y < yPos + chunkTileCount.y; ++y)
239  {
240  for (auto x = xPos; x < xPos + chunkTileCount.x; ++x)
241  {
242  if (idx < m_chunkTileIDs.size() && m_chunkTileIDs[idx].ID >= ca->m_firstGID
243  && m_chunkTileIDs[idx].ID <= ca->m_lastGID)
244  {
245  if (registerAnimation && m_animTiles.find(m_chunkTileIDs[idx].ID) != m_animTiles.end())
246  {
247  AnimationState as;
248  as.animTile = m_animTiles[m_chunkTileIDs[idx].ID];
249  as.startTime = sf::milliseconds(0);
250  as.tileCords = sf::Vector2u(x,y);
251  m_activeAnimations.push_back(as);
252  }
253 
254  sf::Vector2f tileOffset(x * mapTileSize.x, (float)y * mapTileSize.y + mapTileSize.y - ca->tileSetSize.y);
255 
256  auto idIndex = m_chunkTileIDs[idx].ID - ca->m_firstGID;
257  sf::Vector2f tileIndex(idIndex % ca->tsTileCount.x, idIndex / ca->tsTileCount.x);
258  tileIndex.x *= ca->tileSetSize.x;
259  tileIndex.y *= ca->tileSetSize.y;
260  Tile tile =
261  {
262 #ifndef __ANDROID__
263  sf::Vertex(tileOffset - getPosition(), m_chunkColors[idx], tileIndex),
264  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(ca->tileSetSize.x, 0.f), m_chunkColors[idx], tileIndex + sf::Vector2f(ca->tileSetSize.x, 0.f)),
265  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(ca->tileSetSize.x, ca->tileSetSize.y), m_chunkColors[idx], tileIndex + sf::Vector2f(ca->tileSetSize.x, ca->tileSetSize.y)),
266  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(0.f, ca->tileSetSize.y), m_chunkColors[idx], tileIndex + sf::Vector2f(0.f, ca->tileSetSize.y))
267 #endif
268 #ifdef __ANDROID__
269  sf::Vertex(tileOffset - getPosition(), m_chunkColors[idx], tileIndex),
270  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(ca->tileSetSize.x, 0.f), m_chunkColors[idx], tileIndex + sf::Vector2f(ca->tileSetSize.x, 0.f)),
271  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(ca->tileSetSize.x, ca->tileSetSize.y), m_chunkColors[idx], tileIndex + sf::Vector2f(ca->tileSetSize.x, ca->tileSetSize.y)),
272  sf::Vertex(tileOffset - getPosition(), m_chunkColors[idx], tileIndex),
273  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(0.f, ca->tileSetSize.y), m_chunkColors[idx], tileIndex + sf::Vector2f(0.f, ca->tileSetSize.y)),
274  sf::Vertex(tileOffset - getPosition() + sf::Vector2f(ca->tileSetSize.x, ca->tileSetSize.y), m_chunkColors[idx], tileIndex + sf::Vector2f(ca->tileSetSize.x, ca->tileSetSize.y))
275 #endif
276  };
277  doFlips(m_chunkTileIDs[idx].flipFlags,&tile[0].texCoords,&tile[1].texCoords,&tile[2].texCoords,&tile[3].texCoords);
278  ca->addTile(tile);
279  }
280  idx++;
281  }

353  void doFlips(std::uint8_t bits, sf::Vector2f *v0, sf::Vector2f *v1, sf::Vector2f *v2, sf::Vector2f *v3)
354  {
355  //0000 = no change
356  //0100 = vertical = swap y axis
357  //1000 = horizontal = swap x axis
358  //1100 = horiz + vert = swap both axes = horiz+vert = rotate 180 degrees
359  //0010 = diag = rotate 90 degrees right and swap x axis
360  //0110 = diag+vert = rotate 270 degrees right
361  //1010 = horiz+diag = rotate 90 degrees right
362  //1110 = horiz+vert+diag = rotate 90 degrees right and swap y axis
363  if(!(bits & tmx::TileLayer::FlipFlag::Horizontal) &&
364  !(bits & tmx::TileLayer::FlipFlag::Vertical) &&
365  !(bits & tmx::TileLayer::FlipFlag::Diagonal) )
366  {
367  //Shortcircuit tests for nothing to do
368  return;
369  }
370  else if(!(bits & tmx::TileLayer::FlipFlag::Horizontal) &&
371  (bits & tmx::TileLayer::FlipFlag::Vertical) &&
372  !(bits & tmx::TileLayer::FlipFlag::Diagonal) )
373  {
374  //0100
375  flipY(v0,v1,v2,v3);
376  }
377  else if((bits & tmx::TileLayer::FlipFlag::Horizontal) &&
378  !(bits & tmx::TileLayer::FlipFlag::Vertical) &&
379  !(bits & tmx::TileLayer::FlipFlag::Diagonal) )
380  {
381  //1000
382  flipX(v0,v1,v2,v3);
383  }
384  else if((bits & tmx::TileLayer::FlipFlag::Horizontal) &&
385  (bits & tmx::TileLayer::FlipFlag::Vertical) &&
386  !(bits & tmx::TileLayer::FlipFlag::Diagonal) )
387  {
388  //1100
389  flipY(v0,v1,v2,v3);
390  flipX(v0,v1,v2,v3);
391  }
392  else if(!(bits & tmx::TileLayer::FlipFlag::Horizontal) &&
393  !(bits & tmx::TileLayer::FlipFlag::Vertical) &&
394  (bits & tmx::TileLayer::FlipFlag::Diagonal) )
395  {
396  //0010
397  flipD(v0,v1,v2,v3);
398  }
399  else if(!(bits & tmx::TileLayer::FlipFlag::Horizontal) &&
400  (bits & tmx::TileLayer::FlipFlag::Vertical) &&
401  (bits & tmx::TileLayer::FlipFlag::Diagonal) )
402  {
403  //0110
404  flipX(v0,v1,v2,v3);
405  flipD(v0,v1,v2,v3);
406  }
407  else if((bits & tmx::TileLayer::FlipFlag::Horizontal) &&
408  !(bits & tmx::TileLayer::FlipFlag::Vertical) &&
409  (bits & tmx::TileLayer::FlipFlag::Diagonal) )
410  {
411  //1010
412  flipY(v0,v1,v2,v3);
413  flipD(v0,v1,v2,v3);
414  }
415  else if((bits & tmx::TileLayer::FlipFlag::Horizontal) &&
416  (bits & tmx::TileLayer::FlipFlag::Vertical) &&
417  (bits & tmx::TileLayer::FlipFlag::Diagonal) )
418  {
419  //1110
420  flipY(v0,v1,v2,v3);
421  flipX(v0,v1,v2,v3);
422  flipD(v0,v1,v2,v3);
423  }

498  Chunk::Ptr& getChunkAndTransform(int x, int y, sf::Vector2u& chunkRelative)
499  {
500  uint32_t chunkX = floor((x * m_MapTileSize.x) / m_chunkSize.x);
501  uint32_t chunkY = floor((y * m_MapTileSize.y) / m_chunkSize.y);
502  chunkRelative.x = ((x * m_MapTileSize.x) - chunkX * m_chunkSize.x ) / m_MapTileSize.x ;
503  chunkRelative.y = ((y * m_MapTileSize.y) - chunkY * m_chunkSize.y ) / m_MapTileSize.y ;
504  return m_chunks[chunkX + chunkY * m_chunkCount.x];

506  void createChunks(const tmx::Map& map, const tmx::TileLayer& layer)
507  {
508  //look up all the tile sets and load the textures
509  const auto& tileSets = map.getTilesets();
510  const auto& layerIDs = layer.getTiles();
511  std::uint32_t maxID = std::numeric_limits<std::uint32_t>::max();
512  std::vector<const tmx::Tileset*> usedTileSets;
513 
514  for (auto i = tileSets.rbegin(); i != tileSets.rend(); ++i)
515  {
516  for (const auto& tile : layerIDs)
517  {
518  if (tile.ID >= i->getFirstGID() && tile.ID < maxID)
519  {
520  usedTileSets.push_back(&(*i));
521  break;
522  }
523  }
524  maxID = i->getFirstGID();
525  }
526 
527  sf::Image fallback;
528  fallback.create(2, 2, sf::Color::Magenta);
529  for (const auto& ts : usedTileSets)
530  {
531  const auto& path = ts->getImagePath();
532  //std::unique_ptr<sf::Texture> newTexture = std::make_unique<sf::Texture>();
533  std::unique_ptr<sf::Texture> newTexture = std::make_unique<sf::Texture>();
534  sf::Image img;
535  if (!img.loadFromFile(path))
536  {
537  newTexture->loadFromImage(fallback);
538  }
539  else
540  {
541  if (ts->hasTransparency())
542  {
543  auto transparency = ts->getTransparencyColour();
544  img.createMaskFromColor({ transparency.r, transparency.g, transparency.b, transparency.a });
545  }
546  newTexture->loadFromImage(img);
547  }
548  m_textureResource.insert(std::make_pair(path, std::move(newTexture)));
549  }
550 
551  //calculate the number of chunks in the layer
552  //and create each one
553  const auto bounds = map.getBounds();
554  m_chunkCount.x = static_cast<sf::Uint32>(std::ceil(bounds.width / m_chunkSize.x));
555  m_chunkCount.y = static_cast<sf::Uint32>(std::ceil(bounds.height / m_chunkSize.y));
556 
557  sf::Vector2u tileSize(map.getTileSize().x, map.getTileSize().y);
558 
559  for (auto y = 0u; y < m_chunkCount.y; ++y)
560  {
561  sf::Vector2f tileCount(m_chunkSize.x / tileSize.x, m_chunkSize.y / tileSize.y);
562  for (auto x = 0u; x < m_chunkCount.x; ++x)
563  {
564  // calculate size of each Chunk (clip against map)
565  if ((x + 1) * m_chunkSize.x > bounds.width)
566  {
567  tileCount.x = (bounds.width - x * m_chunkSize.x) / map.getTileSize().x;
568  }
569  if ((y + 1) * m_chunkSize.y > bounds.height)
570  {
571  tileCount.y = (bounds.height - y * m_chunkSize.y) / map.getTileSize().y;
