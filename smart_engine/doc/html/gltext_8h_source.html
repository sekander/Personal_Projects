355 GLT_API void gltViewport(GLsizei width, GLsizei height)
356 {
357  _GLT_ASSERT(width > 0);
358  _GLT_ASSERT(height > 0);
359 
360  const GLfloat left = 0.0f;
361  const GLfloat right = (GLfloat)width;
362  const GLfloat bottom = (GLfloat)height;
363  const GLfloat top = 0.0f;
364  const GLfloat zNear = -1.0f;
365  const GLfloat zFar = 1.0f;
366 
367  const GLfloat projection[16] = {
368  (2.0f / (right - left)), 0.0f, 0.0f, 0.0f,
369  0.0f, (2.0f / (top - bottom)), 0.0f, 0.0f,
370  0.0f, 0.0f, (-2.0f / (zFar - zNear)), 0.0f,
371 
372  -((right + left) / (right - left)),
373  -((top + bottom) / (top - bottom)),
374  -((zFar + zNear) / (zFar - zNear)),
375  1.0f,
376  };
377 
378  memcpy(_gltText2DProjectionMatrix, projection, 16 * sizeof(GLfloat));
379 }

658 {

673 GLT_API void _gltUpdateBuffers(GLTtext *text)
674 {
675  if (!text || !text->_dirty)
676  return;
677 
678  if (text->_vertices)
679  {
680  text->vertexCount = 0;
681 
682  free(text->_vertices);
683  text->_vertices = GLT_NULL;
684  }
685 
686  if (!text->_text || !text->_textLength)
687  {
688  text->_dirty = GL_FALSE;
689  return;
690  }
691 
692  const GLsizei countDrawable = gltCountDrawableCharacters(text->_text);
693 
694  if (!countDrawable)
695  {
696  text->_dirty = GL_FALSE;
697  return;
698  }
699 
700  const GLsizei vertexCount = countDrawable * 2 * 3; // 3 vertices in a triangle, 2 triangles in a quad
701 
702  const GLsizei vertexSize = _GLT_TEXT2D_VERTEX_SIZE;
703  GLfloat *vertices = (GLfloat*)malloc(vertexCount * vertexSize * sizeof(GLfloat));
704 
705  if (!vertices)
706  return;
707 
708  GLsizei vertexElementIndex = 0;
709 
710  GLfloat glyphX = 0.0f;
711  GLfloat glyphY = 0.0f;
712 
713  GLfloat glyphWidth;
714  const GLfloat glyphHeight = (GLfloat)_gltFontGlyphHeight;
715 
716  const GLfloat glyphAdvanceX = 0.0f;
717  const GLfloat glyphAdvanceY = 0.0f;
718 
719  _GLTglyph glyph;
720 
721  char c;
722  int i;
723  for (i = 0; i < text->_textLength; i++)
724  {
725  c = text->_text[i];
726 
727  if (c == '\n')
728  {
729  glyphX = 0.0f;
730  glyphY += glyphHeight + glyphAdvanceY;
731 
732  continue;
733  }
